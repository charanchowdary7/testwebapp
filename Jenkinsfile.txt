#Jenkinsfile for clone build test & push image to docker hub.
node {
       def app
       stage('Clone repository') {
       checkout scm
       }

       stage('Build image') {
       app = docker.build("charanchowdary7/testwebapp")
       }
    
       stage('Test image') {
       app.inside { 
       sh 'echo "Tests passed" '
       }
 
 }

       stage('push image') {
       docker.withRegistry('https://registry.hub.docker.com', 'docker-hub-credentials') {
       app.push("${env.BUILD_NUMBER}")
       app.push("latest")
      }
   }
}
